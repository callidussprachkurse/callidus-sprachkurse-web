 * Zweck: Verschickt bei JEDER Einsendung des Google-Formulars "KURSANMELDUNG"
 * automatisch eine Bestätigungsmail an den Anmeldenden — mit übersichtlicher
 * Tabelle seiner Angaben UND dem AGB-/Datenschutz-PDF. Erfüllt "AGB im Postfach".

 * SPRACHE: Website schreibt "LANG: DE"/"LANG: EN" in die Büro-Notiz → Mail entsprechend.
 * Plain-Text-Teil als Fallback, HTML-Tabelle für die Anzeige.
 *
 * EINRICHTUNG: Trigger onAnmeldung (Aus Formular / Bei Formularübermittlung).

var AGB_FILE_ID   = "13U_Zca4nsAkmLlpjBDaLFdfVr42ibmHa"; // CALLIDUS-AGB-und-Datenschutz.pdf in eurem Drive
var ABSENDER_NAME = "CALLIDUS Sprachkurse";

var KOPIE_AN      = "info@callidus-sprachkurse.de";      // stille Kopie (Nachweis)

function _esc(s) {
  return String(s == null ? "" : s).replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;");
}

function onAnmeldung(e) {
  var resp = e.response;
  var email = resp.getRespondentEmail();
  if (!email) return;

  function feld(teil) {
    var items = resp.getItemResponses();
    for (var i = 0; i < items.length; i++) {
      if (items[i].getItem().getTitle().indexOf(teil) > -1) return items[i].getResponse() || "";
    }
    return "";
  }
  var vorname    = feld("Vorname");
  var nachname   = feld("Nachname");
  var geburt     = feld("Geburtsdatum");
  var anschrift  = feld("Anschrift");
  var telefon    = feld("Telefon");
  var kurs       = feld("Welche Kurse");
  var nachricht  = feld("Nachricht");
  var geschlecht = feld("Geschlecht");

  var det = {};
  String(nachricht).split("\n").forEach(function (line) {
    var idx = line.indexOf(":");
    if (idx > 0) det[line.substring(0, idx).trim()] = line.substring(idx + 1).trim();
  });

  var EN = String(det["LANG"] || "DE").toUpperCase() === "EN";
  var name = (vorname + " " + nachname).trim();

  var L = EN
    ? { name:"Name", kurs:"Course", termin:"Date", vk:"Prior knowledge", g:"Gender", geb:"Date of birth", adr:"Address", tel:"Phone", mail:"Email", anm:"Note" }
    : { name:"Name", kurs:"Kurs", termin:"Termin", vk:"Vorkenntnisse", g:"Geschlecht", geb:"Geburtsdatum", adr:"Anschrift", tel:"Telefon", mail:"E-Mail", anm:"Anmerkung" };

  // Zeilen als [Label, Wert]
  var rows = [];
  if (name) rows.push([L.name, name]);
  rows.push([L.kurs, det["Kurs"] || kurs || "—"]);
  if (det["Termin/Wunsch"]) rows.push([L.termin, det["Termin/Wunsch"]]);
  if (det["Vorkenntnisse"]) rows.push([L.vk, det["Vorkenntnisse"]]);
  var gWert = geschlecht || det["Geschlecht"];
  if (gWert) { if (EN) { var gm = {"männlich":"male","weiblich":"female","divers":"diverse"}; gWert = gm[gWert] || gWert; } rows.push([L.g, gWert]); }
  if (geburt)    rows.push([L.geb, geburt]);
  if (anschrift) rows.push([L.adr, String(anschrift).replace(/\n/g, ", ")]);
  if (telefon)   rows.push([L.tel, telefon]);
  rows.push([L.mail, email]);
  if (det["Anmerkung"]) rows.push([L.anm, det["Anmerkung"]]);

  // Texte
  var T = EN ? {
    betreff: "Your registration with CALLIDUS Sprachkurse Hamburg",
    onbehalf: "On behalf of the Italian Cultural Institute Hamburg",
    greeting: vorname ? ("Hello " + vorname + ",") : "Hello,",
    intro: "thank you for registering " + ((det["Kurs"] || kurs) ? ("for " + (det["Kurs"] || kurs)) : "for your course") + ". Attached you will find our Terms and Conditions and Privacy Policy (in German).",
    section: "Your registration",
    binding: "We will send you your invoice shortly (course materials already included). Your registration is hereby binding.",
    payment: "Please note: you are officially listed as a course participant only after full payment has been received.",
    contact: "Questions about your course: info@callidus-sprachkurse.de · about your invoice: rechnungsstelle@callidus-magdeburg.de",
    wrTitle: "Right of withdrawal",
    wrNote: " (This is a courtesy translation. The legally binding version is the German one.)",
    wr: [
      "You have the right to withdraw from this contract within fourteen days without giving any reason. The withdrawal period is fourteen days from the day the contract was concluded. To exercise your right of withdrawal, you must inform us (CALLIDUS Privatinstitut für Lernförderung e.K., owner Danny Matz, Harnackstraße 5, 39104 Magdeburg, email: info@callidus-magdeburg.de) of your decision by means of a clear statement (e.g. a letter or email). To meet the deadline, it is sufficient to send your notification before the period expires.",
      "Consequences of withdrawal: We will refund all payments without undue delay, at the latest within fourteen days of receiving your withdrawal. If you requested that the course begin during the withdrawal period, you owe us a reasonable amount for the services already provided up to the point of withdrawal (pro rata).",
      "Model withdrawal form: To CALLIDUS Privatinstitut für Lernförderung e.K., Harnackstraße 5, 39104 Magdeburg, info@callidus-magdeburg.de — I hereby withdraw from the contract for the following course: __________, concluded on: __________, name: __________, address: __________, date: __________."
    ],
    signoff: "Warm regards",
    team: "Your CALLIDUS goes Italiano team"
  } : {
    betreff: "Ihre Anmeldung bei CALLIDUS Sprachkurse Hamburg",
    onbehalf: "Im Auftrag des Italienischen Kulturinstituts Hamburg",
    greeting: vorname ? ("Hallo " + vorname + ",") : "Hallo,",
    intro: "vielen Dank für Ihre Anmeldung " + ((det["Kurs"] || kurs) ? ("für " + (det["Kurs"] || kurs)) : "für Ihren Kurs") + ". Im Anhang finden Sie unsere Allgemeinen Geschäftsbedingungen und die Datenschutzerklärung.",
    section: "Ihre Anmeldung",
    binding: "Wir senden Ihnen in Kürze Ihre Rechnung (Lehrmaterial bereits inklusive). Ihre Anmeldung ist hiermit verbindlich.",
    payment: "Bitte beachten Sie: Offiziell als Kursteilnehmer/in gelistet sind Sie erst nach vollständigem Zahlungseingang.",
    contact: "Bei Fragen zu Ihrem Kurs: info@callidus-sprachkurse.de · zu Ihrer Rechnung: rechnungsstelle@callidus-magdeburg.de",
    wrTitle: "Widerrufsbelehrung",
    wrNote: "",
    wr: [
      "Widerrufsrecht: Sie haben das Recht, binnen vierzehn Tagen ohne Angabe von Gründen diesen Vertrag zu widerrufen. Die Widerrufsfrist beträgt vierzehn Tage ab dem Tag des Vertragsabschlusses. Um Ihr Widerrufsrecht auszuüben, müssen Sie uns (CALLIDUS Privatinstitut für Lernförderung e.K., Inhaber Danny Matz, Harnackstraße 5, 39104 Magdeburg, E-Mail: info@callidus-magdeburg.de) mittels einer eindeutigen Erklärung (z. B. Brief oder E-Mail) über Ihren Entschluss informieren. Zur Wahrung der Frist genügt die rechtzeitige Absendung.",
      "Folgen des Widerrufs: Wir erstatten Ihnen alle Zahlungen unverzüglich, spätestens binnen vierzehn Tagen ab Zugang Ihres Widerrufs. Haben Sie verlangt, dass der Kurs während der Widerrufsfrist beginnt, schulden Sie uns einen angemessenen Betrag für die bis zum Widerruf bereits erbrachten Leistungen (anteilig).",
      "Muster-Widerrufsformular: An CALLIDUS Privatinstitut für Lernförderung e.K., Harnackstraße 5, 39104 Magdeburg, info@callidus-magdeburg.de — Hiermit widerrufe ich den Vertrag über folgenden Kurs: __________, abgeschlossen am: __________, Name: __________, Anschrift: __________, Datum: __________."
    ],
    signoff: "Mit herzlichen Grüßen",
    team: "Ihr Team von CALLIDUS goes Italiano"
  };

  // Plain-Text (Fallback)
  var plain = T.greeting + "\n\n" + T.intro + "\n\n" + T.section + ":\n" +
    rows.map(function (r) { return r[0] + ": " + r[1]; }).join("\n") + "\n\n" +
    T.binding + "\n" + T.payment + "\n\n" + T.contact + "\n\n" +
    "— " + T.wrTitle + " —" + T.wrNote + "\n" + T.wr.join("\n") + "\n\n" +
    T.signoff + "\n" + T.team;

  // HTML-Tabelle (Markenfarbe Messing/Creme)
  var lbl = "padding:8px 12px;background:#F3ECDD;border:1px solid #E2D7C3;font-weight:600;color:#8C6938;white-space:nowrap;vertical-align:top;width:40%;";
  var val = "padding:8px 12px;border:1px solid #E2D7C3;vertical-align:top;color:#2b2b2b;";
  var subc = "padding:10px 12px 6px;font-weight:700;color:#8C6938;background:#EDE2CD;border:1px solid #E2D7C3;";
  var trs = '<tr><td colspan="2" style="' + subc + '">' + _esc(T.section) + "</td></tr>";
  rows.forEach(function (r) {
    trs += '<tr><td style="' + lbl + '">' + _esc(r[0]) + '</td><td style="' + val + '">' + _esc(r[1] || "—") + "</td></tr>";
  });
  var html =
    '<div style="font-family:Arial,Helvetica,sans-serif;font-size:14px;color:#2b2b2b;line-height:1.6;max-width:600px;margin:0 auto;">' +
    '<p style="margin:0 0 2px;font-size:17px;color:#16130F;"><strong>CALLIDUS Sprachkurse Hamburg</strong></p>' +
    '<p style="margin:0 0 16px;font-size:12px;color:#8C6938;letter-spacing:.04em;">' + _esc(T.onbehalf) + "</p>" +
    '<p style="margin:0 0 14px;">' + _esc(T.greeting) + "<br>" + _esc(T.intro) + "</p>" +
    '<table style="border-collapse:collapse;width:100%;font-size:14px;">' + trs + "</table>" +
    '<p style="margin:16px 0 0;">' + _esc(T.binding) + '<br><span style="color:#6E665A;font-size:13px;">' + _esc(T.payment) + "</span></p>" +
    '<p style="margin:12px 0 0;font-size:13px;color:#5c5446;">' + _esc(T.contact) + "</p>" +
    '<div style="margin:16px 0 0;padding:12px 14px;background:#FAF6EE;border:1px solid #E2D7C3;border-radius:6px;font-size:12px;color:#5c5446;line-height:1.5;">' +
    '<strong style="color:#8C6938;">' + _esc(T.wrTitle) + "</strong>" + _esc(T.wrNote) + "<br>" + T.wr.map(_esc).join("<br><br>") + "</div>" +
    '<p style="margin:16px 0 0;">' + _esc(T.signoff) + "<br><strong>" + _esc(T.team) + "</strong></p>" +
    "</div>";

  var agb = DriveApp.getFileById(AGB_FILE_ID).getBlob();
  GmailApp.sendEmail(email, T.betreff, plain, {
    name: ABSENDER_NAME,
    replyTo: ANTWORT_AN,
    bcc: KOPIE_AN,
    htmlBody: html,
    attachments: [agb]
  });
}

/**
 * EINMALIG ausführen: fügt dem Formular die Pflichtfrage "Geschlecht" hinzu.
 */
function addGeschlecht() {
  var form = FormApp.getActiveForm();
  var items = form.getItems();
  for (var i = 0; i < items.length; i++) {
    if (items[i].getTitle() === "Geschlecht") return "existiert bereits";
  }
  var gebIndex = -1;
  for (var i = 0; i < items.length; i++) {
    if (items[i].getTitle().indexOf("Geburtsdatum") > -1) { gebIndex = items[i].getIndex(); break; }
  }
  var q = form.addMultipleChoiceItem();
  q.setTitle("Geschlecht").setChoiceValues(["männlich", "weiblich", "divers"]).setRequired(true);
  if (gebIndex > -1) form.moveItem(q.getIndex(), gebIndex + 1);
  return "hinzugefügt, itemId=" + q.getId();
}
